import os
import sys
import django

sys.path.append("PROJECT_ROOT")  # Add Project-root to sys-path.
os.environ.setdefault("DJANGO_SETTINGS_MODULE", "PROJECT_NAME.settings") # Load Project-config
django.setup() # Django Setup
from app_name.models import Person  # Load Project-App-Model 
persons = Person.objects.all() # Get all rows.
